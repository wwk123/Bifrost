  const activeMembersCount = team.members.filter(m =>
    m.contribution.weeklyEarnings > 0

  const activityRate = activeMembersCount / team.members.length;
  const synergyBonus = activityRate * 0.15; // æœ€é«˜ +15%
  totalMultiplier += synergyBonus;

  // 3. æˆå°±åŠ æˆï¼ˆå›¢é˜Ÿæ€»æˆå°±æ•°ï¼‰
  const totalAchievements = team.members.reduce(
    (sum, m) => sum + (m.achievements?.length || 0),

  const achievementBonus = Math.min(
    totalAchievements * 0.005,
    0.25
  ); // æ¯ä¸ªæˆå°± +0.5%ï¼Œä¸Šé™ +25%
  totalMultiplier += achievementBonus;

  // 4. é˜Ÿé•¿åŠ æˆï¼ˆå¦‚æœé˜Ÿé•¿æ˜¯å‰ 10 åï¼‰
  const captain = team.members.find(m => m.role === 'captain');
